Ex 4b.

Algo du pseudocode :

string text = gettext(); # obtient le text d'un fichier.

char[] tokeniser(string text){
	char [] tokens = []; 
	string buffer = ""; # string vide
	for char in text:
		if char not in [" ", "\n", "\t"]:
			buffer = buffer + char;
		else : 
			tokens.push(buffer);
			buffer = "";
	return tokens;
}
